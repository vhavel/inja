16 inline nonstd::string_view convert_dot_to_json_pointer(nonstd::string_view dot, std::string& out) {

61  const json* get_imm(const Bytecode& bc) {
62  std::string ptr_buffer;
63  nonstd::string_view ptr;
64  switch (bc.flags & Bytecode::Flag::ValueMask) {
65  case Bytecode::Flag::ValuePop:
66  return nullptr;
67  case Bytecode::Flag::ValueImmediate:
68  return &bc.value;
69  case Bytecode::Flag::ValueLookupDot:
70  ptr = convert_dot_to_json_pointer(bc.str, ptr_buffer);
71  break;
72  case Bytecode::Flag::ValueLookupPointer:
73  ptr_buffer += '/';
74  ptr_buffer += bc.str;
75  ptr = ptr_buffer;
76  break;
77  }
78  try {
79  return &m_data->at(json::json_pointer(ptr.data()));
80  } catch (std::exception&) {
81  // try to evaluate as a no-argument callback
82  if (auto callback = m_callbacks.find_callback(bc.str, 0)) {
83  std::vector<const json*> arguments {};
84  m_tmp_val = callback(arguments);
85  return &m_tmp_val;
86  }
87  inja_throw("render_error", "variable '" + static_cast<std::string>(bc.str) + "' not found");
88  return nullptr;
89  }

172 
173  switch (bc.op) {
174  case Bytecode::Op::Nop: {
175  break;
176  }
177  case Bytecode::Op::PrintText: {
178  os << bc.str;
179  break;
180  }
181  case Bytecode::Op::PrintValue: {
182  const json& val = *get_args(bc)[0];
183  if (val.is_string()) {

198  pop_args(bc);
199  m_stack.emplace_back(std::move(result));
200  break;
201  }
202  case Bytecode::Op::Lower: {
203  auto result = get_args(bc)[0]->get<std::string>();
204  std::transform(result.begin(), result.end(), result.begin(), ::tolower);
205  pop_args(bc);
206  m_stack.emplace_back(std::move(result));
207  break;
208  }
209  case Bytecode::Op::Range: {
210  int number = get_args(bc)[0]->get<int>();
211  std::vector<int> result(number);
212  std::iota(std::begin(result), std::end(result), 0);
213  pop_args(bc);
214  m_stack.emplace_back(std::move(result));
215  break;
216  }
217  case Bytecode::Op::Length: {
218  const json& val = *get_args(bc)[0];
219

223  } else {

225  }

228  m_stack.emplace_back(result);
229  break;
230  }
231  case Bytecode::Op::Sort: {
232  auto result = get_args(bc)[0]->get<std::vector<json>>();
233  std::sort(result.begin(), result.end());
234  pop_args(bc);
235  m_stack.emplace_back(std::move(result));
236  break;
237  }
238  case Bytecode::Op::At: {
239  auto args = get_args(bc);
240  auto result = args[0]->at(args[1]->get<int>());
241  pop_args(bc);
242  m_stack.emplace_back(result);
243  break;
244  }

247  pop_args(bc);

249  break;
250  }

253  pop_args(bc);

255  break;
256  }

260  int precision = args[1]->get<int>();
261  pop_args(bc);
262  m_stack.emplace_back(std::round(number * std::pow(10.0, precision)) / std::pow(10.0, precision));
263  break;
264  }

267  int number = args[0]->get<int>();
268  int divisor = args[1]->get<int>();
269  pop_args(bc);

271  break;
272  }
273  case Bytecode::Op::Odd: {

275  pop_args(bc);

277  break;

365  pop_args(bc);

367  break;

561  break;
562  }
